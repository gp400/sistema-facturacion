create database ProgWeb;

create table Integrantes (
	id int auto_increment,
	documento varchar(20), -- cedula/pasaporte
	foto text,
	nombre varchar(50),
	apellido varchar(50),
	sexo varchar(10),
	fechaNacimiento date,
	direccion varchar(100),
	telefono varchar(15),
	email varchar(30),
	estadoCivil char(5),
	tipoSangre varchar(5),
	alergias text,
	comentario text,
	estado int,
    primary key (id)
)
